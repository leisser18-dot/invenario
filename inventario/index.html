async function exportarExcelConImagen() { const workbook = new ExcelJS.Workbook(); const worksheet = workbook.addWorksheet('Inventario Unificado'); // 1. Configuración de columnas según COLUMNAS_OFICIALES worksheet.columns = COLUMNAS_OFICIALES.map(col => ({ header: col.toUpperCase().replace('_', ' '), key: col, width: 20 })); // 2. Integración del Logo y Encabezado try { // Intentamos cargar el logo desde la raíz de tu repositorio const response = await fetch('./logo.png'); if (!response.ok) throw new Error('Logo no encontrado'); const buffer = await response.arrayBuffer(); const logoId = workbook.addImage({ buffer: buffer, extension: 'png', }); // Crear espacio para el encabezado (6 filas) for(let i=0; i<6; i++) worksheet.insertRow(1, []); // Insertar Logo (Ajustado a la esquina superior izquierda) worksheet.addImage(logoId, { tl: { col: 0, row: 0.2 }, ext: { width: 140, height: 70 } }); // Título del reporte alineado a la derecha worksheet.mergeCells('E2:K2'); const titleCell = worksheet.getCell('E2'); titleCell.value = 'DEPARTAMENTO DE SISTEMAS - REPORTE DE INVENTARIO'; titleCell.font = { name: 'Segoe UI', size: 14, bold: true }; titleCell.alignment = { horizontal: 'right' }; worksheet.mergeCells('E3:K3'); const infoCell = worksheet.getCell('E3'); infoCell.value = `FIBEX TELECOM - GENERADO EL: ${new Date().toLocaleDateString()}`; infoCell.font = { name: 'Segoe UI', size: 10 }; infoCell.alignment = { horizontal: 'right' }; } catch (e) { console.warn("No se pudo cargar 'logo.png'. Asegúrate de que esté en la raíz de tu repo GitHub.", e); for(let i=0; i<6; i++) worksheet.insertRow(1, []); } // 3. Estilo de Encabezados (Fila 7) const headerRow = worksheet.getRow(7); headerRow.eachCell((cell) => { cell.font = { bold: true, color: { argb: 'FF000000' } }; cell.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFF2F2F2' } }; cell.alignment = { horizontal: 'center', vertical: 'middle' }; cell.border = { top: { style: 'thin' }, bottom: { style: 'thick' }, // Línea gruesa para separar de los datos left: { style: 'thin' }, right: { style: 'thin' } }; }); // 4. Carga de todos los registros unificados DATA_LOCAL.forEach((eq) => { const row = worksheet.addRow(eq); row.eachCell((cell) => { // Convertimos a MAYÚSCULAS para mantener la estética profesional cell.value = cell.value ? cell.value.toString().toUpperCase() : '-'; cell.font = { name: 'Segoe UI', size: 9 }; cell.border = { bottom: { style: 'hair' }, left: { style: 'thin' }, right: { style: 'thin' } }; }); }); // 5. Generación del archivo const finalBuffer = await workbook.xlsx.writeBuffer(); saveAs(new Blob([finalBuffer]), `Inventario_Unificado_Fibex_${new Date().toISOString().slice(0,10)}.xlsx`); }
